\section{General Functions}
For any two functions $u$ and $v$:
\[ (u + v)' = u' + v' \]
When there is a constant, $c$:
\[ (c \cdot u)' = c \cdot u' \]
\section{Trigonometric Functions}
\begin{align*}
    \frac{d}{dx} \sin x &= \cos x\\
    \frac{d}{dx} \cos x &= - \sin x
\end{align*}
